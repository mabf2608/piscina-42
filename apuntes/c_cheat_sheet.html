Punteros esencial 
int  x = 5;
int *p = &x;   // p apunta a x

*p = 10;       // desreferenciar → x=10
printf("%p", p); // dirección

// Puntero a array
int arr[] = {1,2,3};
int *q = arr;  // = &arr[0]
*(q + 1);      // = arr[1]

// Puntero nulo
int *null_p = NULL;

// Puntero a puntero
int **pp = &p;

Siempre inicializar punteros. Un puntero sin inicializar puede causar comportamiento indefinido.

Memoria dinámica esencial 
// #include <stdlib.h>

// Reservar un int
int *p = (int*) malloc(sizeof(int));

// Array dinámico de n ints
int *arr = (int*) malloc(n * sizeof(int));

// Inicializado a 0
int *z = (int*) calloc(n, sizeof(int));

// Redimensionar
arr = (int*) realloc(arr, m * sizeof(int));

// Liberar (¡siempre!)
free(p);
p = NULL;

Comprobar siempre que malloc no devuelve NULL antes de usar el puntero.

Structs estructuras 
// Definir struct
typedef struct {
  char  nombre[50];
  int   edad;
  float nota;
} Persona;

// Crear e inicializar
Persona p = {"Ana", 21, 9.5f};
p.edad = 22;           // acceso directo

// Puntero a struct
Persona *ptr = &p;
ptr->edad = 23;        // flecha

// Struct dinámico
Persona *q = (Persona*) malloc(sizeof(Persona));
free(q);

Listas enlazadas estructuras avanzadas 
// Nodo
typedef struct Nodo {
  int         valor;
  struct Nodo *siguiente;
} Nodo;

// Crear nodo
Nodo *nuevo(int v) {
  Nodo *n = malloc(sizeof(Nodo));
  n->valor = v;
  n->siguiente = NULL;
  return n;
}

// Recorrer
Nodo *cur = cabeza;
while (cur != NULL) {
  printf("%d\n", cur->valor);
  cur = cur->siguiente;
}

Punteros a función funciones 
// Declarar puntero a función
int (*op)(int, int);

int sumar(int a, int b) { return a+b; }
int restar(int a, int b) { return a-b; }

op = sumar;
printf("%d\n", op(3, 2));  // 5
op = restar;
printf("%d\n", op(3, 2));  // 1

// Como parámetro (ej: qsort)
int cmp(const void *a, const void *b) {
  return (*(int*)a - *(int*)b);
}
qsort(arr, n, sizeof(int), cmp);
